# TEDxFargo Effects — First Cut Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Prove the whole custom-effect pipeline end-to-end on hardware with one effect — the CHARGE sign ignites like neon (letters flickering on C→H→A→R→G→E), driven by a geometry-aware WLED usermod.

**Architecture:** `tools/gen_wiring.py` emits a baked geometry header (`charge_geometry.h`) from the wiring data. A small WLED usermod (`usermods/tedxfargo/`) includes it, defines the `mode_charge_bootup` effect, and registers it via `strip.addEffect()`. The usermod source is versioned in this (CHARGE) repo and copied into a clone of the WLED firmware to build/flash. The device stays in 2D matrix mode; the effect places color at real LEDs with `SEGMENT.setPixelColorXY(col, row, color)`.

**Tech Stack:** Python 3 (header generation + host test), C++ / Arduino / WLED 16.0.1 (usermod + effect), PlatformIO (build/flash), ESP32 (Gledopto GL-C-616WL).

## Global Constraints

Copied from `docs/superpowers/specs/2026-07-16-tedxfargo-effects-design.md`. Every task inherits these.

- **WLED base:** direct clone of `GLEDOPTO/WLED` branch `gledopto-16.0.1` (== WLED 16.0.1). All API facts below were verified against this version.
- **Effect mode signature is `void mode_x()`** in WLED 16.0.1 — NOT `uint16_t` and NOT `return FRAMETIME` (that is older WLED). `typedef void (*mode_ptr)();` (FX.h). Frame timing is derived inside the effect from `strip.now`.
- **Register effects with `strip.addEffect(255, &mode_fn, _data_META)`** — `id == 255` appends to the next free slot; returns the assigned id, or `255` on failure.
- **Effect metadata string:** `"Name@slider1,slider2;colors;palette;flags"`. 2D effects use flags `2`. Verified reference: PS_Comet uses `"...;;!;2;..."`.
- **A usermod is a folder in `usermods/` containing `library.json` + a `.cpp`** with a `Usermod` subclass and a trailing `static Instance x; REGISTER_USERMOD(x);`. Enable it with `custom_usermods = tedxfargo` in `[env:GL-C-616WL]`.
- **Device stays 2D.** Effects assume a full-canvas 2D segment anchored at (0,0), so segment-relative XY equals absolute grid XY. Guard with `SEGMENT.is2D()`.
- **Header ↔ ledmap coupling:** `CHARGE_COL/ROW` MUST match the flashed `wled/word-controller/ledmap.json` (both generated by `gen_wiring.py` at `WORD_CELL_MM = 12.0`). Regenerate and re-flash together.
- **Letter chain ranges (verified):** C 0–61, H 62–142, A 143–214, R 215–295, G 296–378, E 379–458 (contiguous — the header asserts this).
- **Two repos:** this CHARGE repo at the current working dir; the WLED clone at `$WLED` (default `~/workspace/WLED-charge`). Usermod source of truth lives here (`wled/usermods/tedxfargo/`) and is copied to `$WLED/usermods/tedxfargo/` before each build.

"""Host checks for the baked CHARGE geometry: internal consistency + agreement
with the flashed 12mm word ledmap. Run: python3 tools/test_charge_geometry.py"""
import json, os, sys
sys.path.insert(0, os.path.dirname(os.path.abspath(__file__)))
import charge_geometry as cg

def _load_word():
    cuts = ("src/parts/word_cuts_repairs.json"
            if os.path.exists("src/parts/word_cuts_repairs.json")
            else "src/parts/word_cuts.json")
    W = json.load(open(cuts))

    px = json.load(open("src/parts/word_pixmap.json"))["pixels"]
    return px, (wx1 - wx0), W["face_h"], wx0, wy0

def main():
    px, w_mm, h_mm, ox, oy = _load_word()
    geo = cg.compute(px, w_mm, h_mm, ox, oy, 12.0)
    ok = True
    def chk(cond, label):
        nonlocal ok; ok &= bool(cond)
        print(("PASS " if cond else "FAIL ") + label)

    chk(geo["num_pixels"] == 459, "459 pixels")
    chk(geo["letter_start"] == [0, 62, 143, 215, 296, 379], "letter starts C,H,A,R,G,E")
    chk(sum(geo["letter_count"]) == 459, "letter counts sum to 459")
    # contiguous coverage of 0..458
    covered = []
    for L in range(6):
        s = geo["letter_start"][L]; c = geo["letter_count"][L]
        covered += list(range(s, s + c))
    chk(sorted(covered) == list(range(459)), "letters cover 0..458 contiguously")
    chk(all(0 <= v < geo["grid_w"] for v in geo["col"]), "col within grid width")
    chk(all(0 <= v < geo["grid_h"] for v in geo["row"]), "row within grid height")
    chk(all(0 <= v <= 255 for v in geo["height"] + geo["xnorm"]), "height/xnorm in 0..255")

    # agreement with the flashed ledmap: ledmap.map[row*W + col] must equal the physical index
    lm = json.load(open("wled/word-controller/ledmap.json"))
    chk((geo["grid_w"], geo["grid_h"]) == (lm["width"], lm["height"]), "grid matches ledmap dims")
    W = lm["width"]; bad = 0
    for i in range(geo["num_pixels"]):
        if lm["map"][geo["row"][i] * W + geo["col"][i]] != i:
            bad += 1
    chk(bad == 0, "every pixel's (col,row) round-trips through the ledmap (%d mismatches)" % bad)

    print("\n" + ("ALL PASS" if ok else "FAILURES"))
    sys.exit(0 if ok else 1)

```python
#!/usr/bin/env python3
"""Baked geometry for the CHARGE word sign, emitted as a C header for the
tedxfargo WLED usermod. Mirrors tools/gen_wiring.py's ledmap() grid mapping
EXACTLY so COL/ROW agree with the flashed ledmap.json."""
import math

LETTERS = "CHARGE"

def compute(pixels, w_mm, h_mm, ox, oy, cell):
    px = sorted(pixels, key=lambda p: p["chain"])
    n = len(px)
    W = math.ceil(w_mm / cell)
    H = math.ceil(h_mm / cell)
    col = [min(W - 1, int((p["x"] - ox) / cell)) for p in px]
    row = [min(H - 1, int((h_mm - (p["y"] - oy) - 0.001) / cell)) for p in px]  # 0 = top
    letter = [LETTERS.index(p["letter"]) for p in px]
    yext = {}
    for p in px:
        lo, hi = yext.get(p["letter"], (p["y"], p["y"]))
        yext[p["letter"]] = (min(lo, p["y"]), max(hi, p["y"]))
    def height_of(p):
        lo, hi = yext[p["letter"]]
        return 0 if hi == lo else int(255 * (p["y"] - lo) / (hi - lo))  # 0 = base (smallest y)
    height = [height_of(p) for p in px]
    xs = [p["x"] for p in px]
    xlo, xhi = min(xs), max(xs)
    xnorm = [0 if xhi == xlo else int(255 * (p["x"] - xlo) / (xhi - xlo)) for p in px]
    starts, counts = [], []
    for L in LETTERS:
        idx = [i for i, p in enumerate(px) if p["letter"] == L]
        assert idx == list(range(idx[0], idx[-1] + 1)), "letter %s not contiguous in chain order" % L
        starts.append(idx[0]); counts.append(len(idx))
    return {"num_pixels": n, "num_letters": len(LETTERS), "grid_w": W, "grid_h": H,
            "letter": letter, "col": col, "row": row, "height": height, "xnorm": xnorm,
            "letter_start": starts, "letter_count": counts}

def emit_header(geo, path):
    def arr(name, ctype, vals):
        return "const %s %s[%d] PROGMEM = { %s };\n" % (
            ctype, name, len(vals), ", ".join(str(v) for v in vals))
    out = ["// Generated by tools/gen_wiring.py via tools/charge_geometry.py — DO NOT EDIT.\n",
           "// Regenerate with the ledmap (same WORD_CELL_MM) so COL/ROW stay in sync.\n",
           "#pragma once\n",
           "#define CHARGE_NUM_PIXELS  %d\n" % geo["num_pixels"],
           "#define CHARGE_NUM_LETTERS %d\n" % geo["num_letters"],
           "#define CHARGE_GRID_W      %d\n" % geo["grid_w"],
           "#define CHARGE_GRID_H      %d\n" % geo["grid_h"],
           arr("CHARGE_LETTER", "uint8_t", geo["letter"]),
           arr("CHARGE_COL", "uint8_t", geo["col"]),
           arr("CHARGE_ROW", "uint8_t", geo["row"]),
           arr("CHARGE_HEIGHT", "uint8_t", geo["height"]),
           arr("CHARGE_XNORM", "uint8_t", geo["xnorm"]),
           arr("CHARGE_LETTER_START", "uint16_t", geo["letter_start"]),
           arr("CHARGE_LETTER_COUNT", "uint16_t", geo["letter_count"])]
    open(path, "w").write("".join(out))
```

## Task 3: Add and register the `mode_charge_bootup` effect

Add the neon boot-up effect and register it. Deliverable: builds clean and the effect is selectable.

**Files:**
- Modify: `wled/usermods/tedxfargo/tedxfargo.cpp` (CHARGE repo), then re-sync to `$WLED`.

**Interfaces:**
- Consumes: `charge_geometry.h` arrays (`CHARGE_LETTER_START/COUNT`, `CHARGE_COL`, `CHARGE_ROW`, `CHARGE_NUM_LETTERS`).
- Produces: effect named `CHARGE Boot` in the WLED effect list.

- [ ] **Step 1: Replace `tedxfargo.cpp`** with the full effect + registration (all identifiers verified against WLED 16.0.1: `void` mode signature, `strip.now`, `hw_random8()`, `qsub8`, `BLACK` (FX.h), `RGBW32(r,g,b,w)`, `color_fade(color, amount, video)`, `SEGMENT.fill/setPixelColorXY/is2D/speed/intensity`, `SEGENV.step`):

```cpp
#include "wled.h"
#include "charge_geometry.h"

// Effect metadata: Name@slider1,slider2;colors;palette;flags(2=2D)
static const char _data_CHARGE_BOOTUP[] PROGMEM = "CHARGE Boot@Speed,Flicker;;;2";

// Neon boot-up: letters ignite in chain order C->H->A->R->G->E, each flickering
// on before settling to electric cyan; once all lit, hold, then loop.
void mode_charge_bootup() {
  if (!SEGMENT.is2D()) { SEGMENT.fill(BLACK); return; }  // needs the matrix

  uint16_t letterMs = 900 - (SEGMENT.speed * 3);          // ~900..~135 ms per letter
  const uint16_t holdMs = 1500;
  uint32_t cycle = (uint32_t)letterMs * CHARGE_NUM_LETTERS + holdMs;

  if (SEGENV.step == 0 || (strip.now - SEGENV.step) > cycle) SEGENV.step = strip.now;  // (re)start
  uint32_t t = strip.now - SEGENV.step;

  SEGMENT.fill(BLACK);                                    // clear the matrix each frame
  const uint32_t cyan = RGBW32(0, 255, 255, 0);           // electric CHARGE cyan
  uint8_t flicker = SEGMENT.intensity;                    // 0..255 flicker strength

  for (uint8_t L = 0; L < CHARGE_NUM_LETTERS; L++) {
    uint32_t litAt = (uint32_t)letterMs * L;
    if (t < litAt) continue;                               // not yet igniting
    uint32_t age = t - litAt;

    uint8_t bri = 255;
    if (age < letterMs) {                                  // igniting: ramp up with random dips
      uint8_t ramp = (uint8_t)((age * 255) / letterMs);
      uint8_t dip  = (hw_random8() < 128) ? (uint8_t)(hw_random8() % (uint16_t)(flicker + 1)) : 0;
      bri = qsub8(ramp, dip);
    }
    uint32_t col = color_fade(cyan, bri, true);

    uint16_t start = pgm_read_word(&CHARGE_LETTER_START[L]);
    uint16_t count = pgm_read_word(&CHARGE_LETTER_COUNT[L]);
    for (uint16_t k = 0; k < count; k++) {
      uint16_t i = start + k;
      SEGMENT.setPixelColorXY(pgm_read_byte(&CHARGE_COL[i]),
                              pgm_read_byte(&CHARGE_ROW[i]), col);
    }
  }
}

class TedxFargoUsermod : public Usermod {
 public:
  void setup() override {
    uint8_t id = strip.addEffect(255, &mode_charge_bootup, _data_CHARGE_BOOTUP);
    if (id == 255) DEBUG_PRINTLN(F("[tedxfargo] addEffect failed (list full)"));
  }
  void loop() override {}
};

static TedxFargoUsermod tedxfargo;
REGISTER_USERMOD(tedxfargo);
```
